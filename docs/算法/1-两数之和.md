## 暴力解法

```javascript
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function (nums, target) {
  for (let i = 0; i < nums.length; i++) {
    for (let j = 0; j < nums.length; j++) {
      if (nums[i] + nums[j] === target && i !== j) {
        return [i, j]
      }
    }
  }
}
```

时间复杂度 O(n^2)

时间复杂度 O(1)

## 备忘录解法（空间换时间）

```javascript
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function (nums, target) {
  let obj = {}
  for (let i = 0; i < nums.length; i++) {
    let num = nums[i]
    let n = target - num
    if (num in obj) {
      return [i, obj[num]]
    } else {
      obj[n] = i
    }
  }
}
```

时间复杂度 O(n)

时间复杂度 O(1)
