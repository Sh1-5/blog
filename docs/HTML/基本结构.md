```html
<!-- 文档类型声明，告诉浏览器当前页面是HTML5页面 -->
<!DOCTYPE html>
<!-- 根元素，lang属性设置语言（帮助翻译工具确定要使用的翻译规则） -->
<html lang="en">
  <!-- head元素，规定文档相关的配置信息 -->
  <head>
    <!-- 字符集 -->
    <meta charset="UTF-8" />
    <!-- IE适配 -->
    <meta http-equiv="X-UA-Compatible" content="IE=edge" />
    <!-- 移动端适配 -->
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <!-- 网页的标题 -->
    <title>Document</title>
  </head>
  <!-- body元素 -->
  <body></body>
</html>
```
